// ============================================================ // MyBudget — Google Apps Script (GAS) // Paste this into Extensions → Apps Script in your Google Sheet // Then: Deploy → New deployment → Web app → Anyone → Deploy // ============================================================ const SHEET_NAME_EXPENSES = 'Expenses'; const SHEET_NAME_MONTHLY = 'MonthlySummary'; const SHEET_NAME_CONFIG = 'Config'; const SHEET_NAME_AUDIT = 'AuditLog'; // ============================================================ // MAIN ENTRY POINT // ============================================================ function doGet(e) { try { const action = e.parameter.action; const callback = e.parameter.callback; // JSONP support let result; if (action === 'ping') result = { success: true }; else if (action === 'addExpense') result = addExpense(JSON.parse(e.parameter.data)); else if (action === 'deleteExpense') result = deleteExpense(e.parameter.id); else if (action === 'getConfig') result = getConfig(); else if (action === 'getMonth') result = getMonth(e.parameter.year, e.parameter.month); else if (action === 'getMonthlySummary') result = getMonthlySummary(e.parameter.year); else if (action === 'getFinancialYear') result = getFinancialYear(e.parameter.fy); else if (action === 'getAll') result = getAll(); else result = { error: 'Unknown action: ' + action }; // Return JSONP if callback provided, else plain JSON if (callback) { return ContentService .createTextOutput(`${callback}(${JSON.stringify(result)})`) .setMimeType(ContentService.MimeType.JAVASCRIPT); } return jsonResponse(result); } catch(err) { return jsonResponse({ error: err.message }); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; let result; if (action === 'addExpense') result = addExpense(body.expense); else if (action === 'setConfig') result = setConfig(body.key, body.value); else if (action === 'deleteExpense') result = deleteExpense(body.id); else result = { error: 'Unknown action: ' + action }; return jsonResponse(result); } catch(err) { return jsonResponse({ error: err.message }); } } function jsonResponse(data) { const output = ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); return output; } // ============================================================ // SETUP — Run this once manually to create headers // ============================================================ function setupSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // Expenses tab const expSheet = ss.getSheetByName(SHEET_NAME_EXPENSES); if (expSheet.getLastRow() === 0) { expSheet.appendRow([ 'id','date','year','month','financialYear', 'merchant','amount','category','subCategory', 'tag','memo','type','groupId','workRelated' ]); expSheet.getRange(1,1,1,14).setFontWeight('bold'); expSheet.setFrozenRows(1); } // MonthlySummary tab const sumSheet = ss.getSheetByName(SHEET_NAME_MONTHLY); if (sumSheet.getLastRow() === 0) { sumSheet.appendRow([ 'yearMonth','year','month','financialYear', 'totalSpent','txCount','workRelatedTotal', 'foodSpend','transportSpend','billsSpend','shoppingSpend', 'healthSpend','leisureSpend','fandomSpend', 'subscriptionsSpend','giftsSpend','otherSpend' ]); sumSheet.getRange(1,1,1,17).setFontWeight('bold'); sumSheet.setFrozenRows(1); } // Config tab const configSheet = ss.getSheetByName(SHEET_NAME_CONFIG); if (configSheet.getLastRow() === 0) { const defaults = [ ['budget', '2000'], ['zeroGoal', '10'], ['startDate', new Date().toISOString().split('T')[0]], ['ownerEmail', ''], ['reportFrequency', 'monthly'], ['financialYearStart', '07-01'], ['currency', 'AUD'], ]; defaults.forEach(row => configSheet.appendRow(row)); configSheet.getRange(1,1,defaults.length,2).setFontWeight('normal'); } // AuditLog tab const auditSheet = ss.getSheetByName(SHEET_NAME_AUDIT); if (auditSheet.getLastRow() === 0) { auditSheet.appendRow(['timestamp','action','expenseId','status','detail']); auditSheet.getRange(1,1,1,5).setFontWeight('bold'); auditSheet.setFrozenRows(1); } SpreadsheetApp.getUi().alert('✅ MyBudget sheets set up successfully!'); } // ============================================================ // HELPERS // ============================================================ function getExpensesSheet() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME_EXPENSES); } function getSummarySheet() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME_MONTHLY); } function getConfigSheet() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME_CONFIG); } function getAuditSheet() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME_AUDIT); } function calcFinancialYear(dateStr) { // Australian FY: July 1 – June 30 const d = new Date(dateStr + 'T00:00:00'); const y = d.getFullYear(); const m = d.getMonth(); // 0-indexed if (m >= 6) return `FY${y}-${y+1}`; // July–Dec else return `FY${y-1}-${y}`; // Jan–June } function audit(action, expenseId, status, detail) { try { getAuditSheet().appendRow([ new Date().toISOString(), action, expenseId || '', status, detail || '' ]); } catch(e) { /* silent */ } } function sheetToObjects(sheet) { const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { let val = row[i]; // Convert Date objects to YYYY-MM-DD string if (val instanceof Date && !isNaN(val)) { const y = val.getFullYear(); const m = String(val.getMonth() + 1).padStart(2, '0'); const d = String(val.getDate()).padStart(2, '0'); val = `${y}-${m}-${d}`; } obj[h] = val; }); return obj; }); } // ============================================================ // GET: Config // ============================================================ function getConfig() { const rows = sheetToObjects(getConfigSheet()); const config = {}; rows.forEach(r => { config[r.key] = r.value; }); return { success: true, config }; } // ============================================================ // POST: setConfig // ============================================================ function setConfig(key, value) { const sheet = getConfigSheet(); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === key) { sheet.getRange(i+1, 2).setValue(value); return { success: true }; } } // Key not found — append sheet.appendRow([key, value]); return { success: true }; } // ============================================================ // GET: getMonth // ============================================================ function getMonth(year, month) { const rows = sheetToObjects(getExpensesSheet()); const filtered = rows.filter(r => String(r.year) === String(year) && String(r.month) === String(month) ); return { success: true, expenses: filtered }; } // ============================================================ // GET: getMonthlySummary // ============================================================ function getMonthlySummary(year) { const rows = sheetToObjects(getSummarySheet()); const filtered = year ? rows.filter(r => String(r.year) === String(year)) : rows; return { success: true, summaries: filtered }; } // ============================================================ // GET: getFinancialYear // ============================================================ function getFinancialYear(fy) { const rows = sheetToObjects(getExpensesSheet()); const filtered = rows.filter(r => r.financialYear === fy); return { success: true, expenses: filtered }; } // ============================================================ // GET: getAll // ============================================================ function getAll() { const rows = sheetToObjects(getExpensesSheet()); return { success: true, expenses: rows }; } // ============================================================ // POST: addExpense // ============================================================ function addExpense(expense) { const sheet = getExpensesSheet(); const d = expense.date; const dateObj = new Date(d + 'T00:00:00'); const year = dateObj.getFullYear(); const month = dateObj.getMonth() + 1; const fy = calcFinancialYear(d); // Flatten expense into rows const rows = buildExpenseRows(expense, year, month, fy); rows.forEach(row => sheet.appendRow(row)); // Update MonthlySummary updateMonthlySummary(year, month, fy); audit('addExpense', expense.id, 'success', `${expense.merchant} $${expense.amount}`); return { success: true, rowsAdded: rows.length }; } function buildExpenseRows(expense, year, month, fy) { const rows = []; if (expense.type === 'single') { rows.push([ expense.id, expense.date, year, month, fy, expense.merchant, expense.amount, expense.category, expense.subCategory || '', expense.tag || '', expense.memo || '', 'single', expense.id, expense.workRelated ? 'TRUE' : 'FALSE' ]); } else if (expense.type === 'receipt') { (expense.items || []).forEach((item, i) => { rows.push([ `${expense.id}-${i}`, expense.date, year, month, fy, expense.merchant, item.amount, item.category, item.name || '', '', '', 'receipt-item', expense.id, expense.workRelated ? 'TRUE' : 'FALSE' ]); }); } else if (expense.type === 'linked') { (expense.items || []).forEach((item, i) => { rows.push([ `${expense.id}-${i}`, expense.date, year, month, fy, expense.merchant, item.amount, item.category, item.name || '', expense.tag || '', `Bank total: $${expense.amount}`, 'linked-item', expense.id, expense.workRelated ? 'TRUE' : 'FALSE' ]); }); } return rows; } // ============================================================ // POST: deleteExpense // ============================================================ function deleteExpense(id) { const sheet = getExpensesSheet(); const data = sheet.getDataRange().getValues(); const headers = data[0]; const idIdx = headers.indexOf('id'); const groupIdx = headers.indexOf('groupId'); // Delete all rows where id OR groupId matches const toDelete = []; for (let i = data.length - 1; i >= 1; i--) { if (data[i][idIdx] === id || data[i][groupIdx] === id) { toDelete.push(i + 1); // 1-indexed } } toDelete.forEach(rowNum => sheet.deleteRow(rowNum)); // Update summary for affected months // (simplified: just recalculate current month) const today = new Date(); updateMonthlySummary(today.getFullYear(), today.getMonth()+1, calcFinancialYear(today.toISOString().split('T')[0])); audit('deleteExpense', id, 'success', `Deleted ${toDelete.length} rows`); return { success: true, rowsDeleted: toDelete.length }; } // ============================================================ // MonthlySummary — update or insert row for yearMonth // ============================================================ function updateMonthlySummary(year, month, fy) { const expSheet = getExpensesSheet(); const sumSheet = getSummarySheet(); const yearMonth = `${year}-${String(month).padStart(2,'0')}`; // Pull all expense rows for this month const allRows = sheetToObjects(expSheet); const monthRows = allRows.filter(r => String(r.year) === String(year) && String(r.month) === String(month) ); // Totals const catMap = { 'Food':'foodSpend','Transport':'transportSpend','Bills':'billsSpend', 'Shopping':'shoppingSpend','Health and Beauty':'healthSpend', 'Leisure':'leisureSpend','Fandom':'fandomSpend', 'Subscriptions':'subscriptionsSpend','Gifts':'giftsSpend','Other':'otherSpend' }; const totals = { totalSpent: 0, workRelatedTotal: 0, foodSpend:0,transportSpend:0,billsSpend:0,shoppingSpend:0, healthSpend:0,leisureSpend:0,fandomSpend:0, subscriptionsSpend:0,giftsSpend:0,otherSpend:0 }; // Count unique groups for txCount const groups = new Set(); monthRows.forEach(r => { const amt = parseFloat(r.amount) || 0; totals.totalSpent += amt; if (r.workRelated === 'TRUE' || r.workRelated === true) totals.workRelatedTotal += amt; const catKey = catMap[r.category]; if (catKey) totals[catKey] += amt; groups.add(r.groupId || r.id); }); const newRow = [ yearMonth, year, month, fy, totals.totalSpent, groups.size, totals.workRelatedTotal, totals.foodSpend, totals.transportSpend, totals.billsSpend, totals.shoppingSpend, totals.healthSpend, totals.leisureSpend, totals.fandomSpend, totals.subscriptionsSpend, totals.giftsSpend, totals.otherSpend ]; // Find existing row const sumData = sumSheet.getDataRange().getValues(); const ymIdx = sumData[0].indexOf('yearMonth'); for (let i = 1; i < sumData.length; i++) { if (sumData[i][ymIdx] === yearMonth) { sumSheet.getRange(i+1, 1, 1, newRow.length).setValues([newRow]); return; } } // Not found — append sumSheet.appendRow(newRow); }
